Database 이름 : csems

기자재 (equipment)
index (int)	auto_increment
rfid (int)
title (varchar(255))
content (longtext)
img (varchar(255)) 	이미지의 url을 저장한다. (서버에 저장하지 않고 불러옴)
borrwed_member_index (int)	= 0 이라면 대출가능
due_date (timestamp)

책(book)
index (int)	auto_increment
rfid (int)
title (varchar(255))
content (longtext)
img (varchar(255)) 	이미지의 url을 저장한다. (서버에 저장하지 않고 불러옴)
borrwed_member_index (int)	= 0 이라면 대출가능
due_date (timestamp)		직접 설정

회원(member)
index (int)	auto_increment
id (varchar(30)) 아이디
password (varchar(255)) 비밀번호 (mysql password 함수 사용)
std_number(varchar(15))	학번
name (varchar(30))
is_admin (boolean)


로그(action_log)
index (int) auto_increment
type (varchar(20)) 'book', 'equipment'
type_index int
action (varchar(30)) 대출/반납
member_index int
action_date (timestamp)


처음 설치시
create database csems DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;
use csems;

create table `member` (
	`index` int primary key auto_increment,
	`id` varchar(30) unique,
	`password` varchar(255),
	`std_number` varchar(15) unique,
	`name` varchar(30),
	`member_level` int
);

create table `book` (
	`index` int primary key auto_increment,
	`rfid` int unique,
	`title` varchar(255),
	`writer` varchar(30),
	`publisher` varchar(50),
	`content` longtext,
	`img` varchar(255),
	`borrowed_member_index` int,
	`due_date` datetime default current_timestamp
);

create table `equipment` (
	`index` int primary key auto_increment,
	`rfid` int unique,
	`title` varchar(255),
	`content` longtext,
	`img` varchar(255),
	`borrowed_member_index` int,
	`due_date` datetime default current_timestamp
);

create table `action_log` (
	`index` int primary key auto_increment,
	`type` varchar(20),
	`type_index` int,
	`action` varchar(20),
	`member_index` int,
	`action_date` datetime default current_timestamp
);
