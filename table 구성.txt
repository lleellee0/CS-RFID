처음 설치시
create database csems;
use csems;

create table `member` (
	`index` int primary key auto_increment,
	`rfid` int unique,
	`std_number` varchar(15) unique,
	`name` varchar(30),
	`is_admin` boolean
);

create table `book` (
	`index` int primary key auto_increment,
	`rfid` int unique,
	`title` varchar(255),
	`content` longtext,
	`img` varchar(255),
	`borrowed_member_index` int,
	`due_date` datetime default current_timestamp
);

create table `equipment` (
	`index` int primary key auto_increment,
	`rfid` int unique,
	`title` varchar(255),
	`content` longtext,
	`img` varchar(255),
	`borrowed_member_index` int,
	`due_date` datetime default current_timestamp
);
